108int MPI_Gather(const void *sendbuf, int sendcnt, MPI_Datatype sendtype,
109 void *recvbuf, int recvcnt, MPI_Datatype recvtype, int root,
110 MPI_Comm comm)
111{
112 int ier = MPI_SUCCESS0;
113 size_t n = 0, nsent = 0, nrecv = 0;
114 if (!sendbuf && !recvbuf)
115 {
116 return ier;
117 }
118 if (comm == MPI_COMM_NULL0 || root)
119 {
120 ier = MPI_ERR_COMM5;
121 }
122
123 switch (sendtype)
124 {
125 case MPI_CHAR2:
126 n = sizeof(char);
127 break;
128 case MPI_INT1:
129 n = sizeof(int);
130 break;
131 case MPI_UNSIGNED4:
132 n = sizeof(unsigned int);
133 break;
134 case MPI_UNSIGNED_LONG5:
135 n = sizeof(unsigned long);
136 break;
137 case MPI_UNSIGNED_LONG_LONG6:
138 n = sizeof(unsigned long long);
139 break;
140 default:
141 return MPI_ERR_TYPE3;
142 }
143 nsent = n * sendcnt;
144
145 switch (recvtype)
146 {
147 case MPI_CHAR2:
148 nrecv = sizeof(char);
149 break;
150 case MPI_INT1:
151 nrecv = sizeof(int);
152 break;
153 case MPI_UNSIGNED4:

156 case MPI_UNSIGNED_LONG5:
157 nrecv = sizeof(unsigned long);

161 break;
162 default:
163 return MPI_ERR_TYPE3;
164 }
165 nrecv = n * recvcnt;
166
167 if (nrecv != nsent)

169 ier = MPI_ERR_COUNT2;

310int MPI_File_open(MPI_Comm /*comm*/, const char *filename, int amode,
311 MPI_Info /*info*/, MPI_File *fh)
312{
313 std::string mode;
314 if (amode | MPI_MODE_RDONLY1)
315 {
316 mode += "r";
317 }
318 if (amode | MPI_MODE_WRONLY2)
319 {
320 mode += "w";
321 }
322 if (amode | MPI_MODE_APPEND4)
323 {
324 mode += "a";
325 }
326 mode += "b";
327
328 *fh = std::fopen(filename, mode.c_str());
329 if (!*fh)
330 {
331 std::snprintf(mpierrmsg, MPI_MAX_ERROR_STRING512, "File not found: %s",
332 filename);
333 return -1;
334 }
335 return MPI_SUCCESS0;
336}

407int MPI_Reduce(const void *sendbuf, void *recvbuf, int count,
408 MPI_Datatype datatype, MPI_Op op, int root, MPI_Comm comm)
409{
410 switch (datatype)
411 {
412 case MPI_CHAR2:
413 if (op == MPI_SUM0)
414 {
415 char *recvBuffer = reinterpret_cast<char *>(recvbuf);
416 const char *sendBuffer = reinterpret_cast<const char *>(sendbuf);
417 *recvBuffer = std::accumulate(sendBuffer, sendBuffer + count, 0);
418 }
419 break;
420 case MPI_INT1:
421 if (op == MPI_SUM0)
422 {
423 int *recvBuffer = reinterpret_cast<int *>(recvbuf);
424 const int *sendBuffer = reinterpret_cast<const int *>(sendbuf);
425 *recvBuffer = std::accumulate(sendBuffer, sendBuffer + count, 0);
426 }
427 break;
428 case MPI_UNSIGNED4:
429 if (op == MPI_SUM0)
430 {
431 unsigned int *recvBuffer =
432 reinterpret_cast<unsigned int *>(recvbuf);
433 const unsigned int *sendBuffer =
434 reinterpret_cast<const unsigned int *>(sendbuf);
435 *recvBuffer = std::accumulate(sendBuffer, sendBuffer + count, 0);
436 }
437 break;
438 case MPI_UNSIGNED_LONG5:
439 if (op == MPI_SUM0)
440 {
441 unsigned long int *recvBuffer =
442 reinterpret_cast<unsigned long int *>(recvbuf);
443 const unsigned long int *sendBuffer =
444 reinterpret_cast<const unsigned long int *>(sendbuf);
445 *recvBuffer = std::accumulate(sendBuffer, sendBuffer + count, 0);
446 }
447 break;
448 case MPI_UNSIGNED_LONG_LONG6:
449 if (op == MPI_SUM0)
450 {
451 unsigned long long int *recvBuffer =
452 reinterpret_cast<unsigned long long int *>(recvbuf);
453 const unsigned long long int *sendBuffer =
454 reinterpret_cast<const unsigned long long int *>(sendbuf);
455 *recvBuffer =
456 std::accumulate(sendBuffer, sendBuffer + count,
457 static_cast<unsigned long long int>(0));
458 }
459 break;
460 default:
461 return MPI_ERR_TYPE3;
462 }
463
464 return 0;
