881 if (!ioformat->server_ID.value) {

891 LOG(fs, "Version 2 ID values differ\n");
892 }
893 free(tmp.value);
894 }
895 break;
896 }
897 new->next = NULL((void*)0);
898 server_format_count++;
899 if (last == NULL((void*)0)) {
900 fs->lists[0] = new;
901 } else {
902 last->next = new;

905 return ioformat;
906 } else {

908 /* we're a proxy, perhaps the master knows this format */
909 IOFormatRep tmp = get_format_from_master(fs, ioformat);
910 if (tmp == NULL((void*)0)) return NULL((void*)0);

912 /* we got a format, free the local version */
913 if (ioformat->server_ID.value != NULL((void*)0)) free(ioformat->server_ID.value);
914 if (ioformat->server_format_rep != NULL((void*)0)) free(ioformat->server_format_rep);
915 free(ioformat);
916 return tmp;
917 }
918 return NULL((void*)0);
919 }
