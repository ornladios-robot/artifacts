646static char *
647get_buffer(size_p)
648int *size_p;
649{
650 static int file_fd = 0;
651 static char *buffer = NULL((void*)0);
652 char *tmp_buffer;
653 static int last_size = -1;
654 int to_read;
655 int tmp_size;
656 unsigned int beef = 0xdeadbeef;
657 int indicator;
658
659 if (read_file == NULL((void*)0)) exit(1);
660
661 if (loaded_FMcontext == NULL((void*)0)) {
662 loaded_FMcontext = create_local_FMcontext();
663 }
664 if (file_fd == 0) {
665 file_fd = open(read_file, O_RDONLY00|O_BINARY0, 0777);
666 buffer = malloc(1);
667 }
668 if (last_size != -1) {
669 if (memcmp(buffer+last_size, &beef, 4) != 0) {
670 printf("memory overwrite error\n");
671 }
672 }
673 if(read(file_fd, &indicator, 4) != 4) exit(1);
674 indicator = ntohl(indicator);
675 if ((indicator >> 24) == 0x2) {
676 /* got a format coming in */
677 int format_rep_size, format_id_size;
678 char *format_id, *format_rep;
679 if (read(file_fd, &format_rep_size, 4) != 4) exit(1);
680 format_rep_size = ntohl(format_rep_size);
681 format_id_size = indicator & 0xff;
682 format_id = malloc(format_id_size);
683 format_rep = malloc(format_rep_size);
684 if (read(file_fd, format_id, format_id_size) != format_id_size) exit(1);
685 if (read(file_fd, format_rep, format_rep_size) != format_rep_size) exit(1);
686 (void) load_external_format_FMcontext(loaded_FMcontext, format_id,
687 format_id_size, format_rep);
688 if(read(file_fd, &indicator, 4) != 4) exit(1);
689 indicator = ntohl(indicator);
690 free(format_id);
691 }
692 if ((indicator >> 24) != 0x3) printf("BAD!\n");
693 to_read = indicator & 0xffffff;
694 last_size = to_read;
695 buffer = realloc(buffer, to_read+4);
696 tmp_buffer = buffer;
697 while((tmp_size = read(file_fd, tmp_buffer, to_read)) != to_read) {
698 if (tmp_size == 0) {
699 free(buffer);
700 return NULL((void*)0);
701 } else if (tmp_size == -1) {
702 perror("Read failure");
703 free(buffer);
704 return NULL((void*)0);
705 }
706 to_read -= tmp_size;
707 tmp_buffer += tmp_size;
708 }
709 memcpy(buffer+last_size, &beef, 4);
710 if (last_size == 0) {
711 free(buffer);
712 close(file_fd);
713 file_fd = 0;
714 return NULL((void*)0);
715 } else {
716 *size_p = last_size;
717 return buffer;
718 }
719}

1189static int seen_count = 0;

1191static void
1192write_buffer(format, buf, size)
1193FMFormat format;
1194char *buf;
1195int size;
1196{
1197 static int file_fd = 0;
1198 int i;
1199 int written = 0;
1200 int indicator;
1201 if (output_file == NULL((void*)0)) return;
1202
1203 if (file_fd == 0) {
1204 file_fd = open(output_file, O_WRONLY01|O_CREAT0100|O_TRUNC01000|O_BINARY0, 0777);
1205 }
1206 for (i=0; i < seen_count; i++) {
1207 if (format == seen_formats[i]) {
1208 written++;
1209 }
1210 }
1211 if (!written) {
1212 struct {
1213 int indicator;
1214 int format_len;
1215 } format_header;
1216 struct iovec vec[4];
1217 char *server_id;
1218 int id_len;
1219 char *server_rep;
1220 int rep_len;
1221 if (verbose) dump_FMFormat(format);
1222 server_id = get_server_ID_FMformat(format, &id_len);
1223 server_rep = get_server_rep_FMformat(format, &rep_len);
1224
1225 /*
1226 * next_data indicator is a 2 4-byte chunks in network byte order.
1227 * In the first chunk,
1228 * the top byte is 0x2, middle 2 are unused and
1229 * the bottom byte is the size of the format id;
1230 * The second chunk holds the length of the format rep;
1231 */
1232
1233 format_header.indicator = htonl((id_len & 0xff) | 0x2 << 24);
1234 format_header.format_len = htonl(rep_len);
1235
1236 vec[0].iov_len = 8;
1237 vec[0].iov_base = &format_header;
1238 vec[1].iov_len = id_len;
1239 vec[1].iov_base = server_id;
1240 vec[2].iov_len = rep_len;
1241 vec[2].iov_base = server_rep;
1242 vec[3].iov_len = 0;
1243 vec[3].iov_base = NULL((void*)0);
1244 if (writev(file_fd, vec, 3) == -1) {
1245 printf("Writev failed\n");
1246 exit(1);
1247 }
1248 seen_formats[seen_count++] = format;
1249 }
1250
1251 /*
1252 * next_data indicator is a 4-byte chunk in network byte order.
1253 * The top byte is 0x3. The bottom 3 bytes are the size of the data.
1254 */
1255 indicator = htonl((size & 0xffffff) | 0x3 << 24);
1256
1257 if (write(file_fd, &indicator, 4) != 4) {printf("Write 4 failed\n");exit(1);}
1258 if (write(file_fd, buf, size) != size) {printf("Write size failed\n");exit(1);}
