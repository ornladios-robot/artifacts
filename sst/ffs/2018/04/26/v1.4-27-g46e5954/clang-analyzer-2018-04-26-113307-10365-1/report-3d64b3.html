736FMfdump_encoded_data(out, format, data, character_limit)
737void *out;
738FMFormat format;
739void *data;
740int character_limit;
741{
742 int ret;
743 int header_size = format->server_ID.length;
744 if (format->variant) {

746 }
747 header_size += (8 - header_size) & 0x7;
748 data = (char*)data + header_size;
749
750 struct dump_state state;
751 init_dump_state(&state);
752 state.output_limit = character_limit;
753 state.encoded = 1;
754 state.offset_base = data;
755 state.out = (FILE*)out;
756 if (FMdumpVerbose)
757 dump_output(&state, strlen(format->format_name) + 15, "Record type %s :", format->format_name);
758 ret = internal_dump_data(format, data, &state);
759 dump_output(&state, 1, "\n");
760 return ret;
761}

786extern void
787FMfdump_encoded_XML(void *out, FMContext c, void *data, int limit)
788{
789 FMFormat format = FMformat_from_ID(c, data);
790 int header_size = format->server_ID.length;
791 struct dump_state state;
792
793 if (format->variant) {
794 header_size += sizeof(INT4int);
795 }
796 header_size += (8 - header_size) & 0x7;
797 data = (char*)data + header_size;
798
799
800 if (FMhas_XML_info(format)) {
801 FMdump_XML(format, data, 1);
802 return;
803 }
804 init_dump_state(&state);
805 state.output_limit = -1;
806 state.encoded = 1;
807 state.offset_base = data;
808 state.use_XML = 1;
809 state.out = (FILE*)out;
810 dump_output(&state, strlen(format->format_name) + 3, "<%s>\n", format->format_name);
811 internal_dump_data(format, data, &state);
812 dump_output(&state, strlen(format->format_name) + 4, "</%s>\n", format->format_name);
813}
