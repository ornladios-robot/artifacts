781static void handle_request(CMtrans_services svc,
782 ib_conn_data_ptr scd,
783 struct control_message *msg)
784{
785 //handling the request message
786
787 //first read the request message from the socket
788 struct ibv_wc wc;
789 tbuffer *tb;
790 int retval = 0;
791 struct request *req;
792
793 req = &msg->u.req;
794
795 tb = findMemory(scd, scd->sd, svc, req->length);
796 if(tb == NULL((void*)0))
797 {
798 svc->trace_out(scd->sd->cm, "Failed to get memory\n");
799 internal_write_response(svc, scd, NULL((void*)0), 0, 0);
800 goto wait_on_q;
801 }
802
803 scd->tb = tb;
804
805 svc->set_pending_write(scd->conn);
806 internal_write_response(svc, scd, tb, req->length, req->request_ID);
807
808wait_on_q:
809 //now the sender will start the data transfer. When it finishes he will
810 //issue a notification after which the data is in memory
811 retval = waitoncq(scd, scd->sd, svc, scd->sd->recv_cq);
812 if(retval)
813 {
814 svc->trace_out(scd->sd->cm, "Error while waiting\n");
815 return;
816 }
817
818 //now we start polling the cq
819
820 do
821 {
822 svc->trace_out(scd->sd->cm, "CMIB poll cq -start");
823 retval = ibv_poll_cq(scd->sd->recv_cq, 1, &wc);
824 svc->trace_out(scd->sd->cm, "CMIB poll cq -end");
825 if(retval > 0 && wc.status == IBV_WC_SUCCESS && wc.opcode == IBV_WC_RECV)
826 {
827 //cool beans - send completed we can go on with our life
828
829 //issue a reccieve so we don't run out
830 retval = ibv_post_recv(scd->dataqp, &scd->isDone.rwr,
831 &scd->isDone.badrwr);
832 if(retval)
833 {
834 scd->sd->svc->trace_out(scd->sd->cm, "CMib unable to post recv %d\n", retval);
835 }
836
837 break;
838 }
839 else
840 {
841 svc->trace_out(scd->sd->cm, "Error polling for write completion\n");
842 break;
843 }
844
845 svc->trace_out(scd->sd->cm, "CMIB poll cq looping");
846
847 }while(1);
848
849 scd->read_buffer = tb->buf;
850 svc->trace_out(scd->sd->cm, "FIrst 16 bytes of receive buffer (len %d) are %02x %02x %02x %02x %02x %02x %02x %02x %02x %02x %02x %02x %02x %02x %02x %02x \n", req->length, ((unsigned char*)tb->buf->buffer)[0], ((unsigned char*)tb->buf->buffer)[1], ((unsigned char*)tb->buf->buffer)[2], ((unsigned char*)tb->buf->buffer)[3], ((unsigned char*)tb->buf->buffer)[4], ((unsigned char*)tb->buf->buffer)[5], ((unsigned char*)tb->buf->buffer)[6], ((unsigned char*)tb->buf->buffer)[7], ((unsigned char*)tb->buf->buffer)[8], ((unsigned char*)tb->buf->buffer)[9], ((unsigned char*)tb->buf->buffer)[10], ((unsigned char*)tb->buf->buffer)[11], ((unsigned char*)tb->buf->buffer)[12], ((unsigned char*)tb->buf->buffer)[13], ((unsigned char*)tb->buf->buffer)[14], ((unsigned char*)tb->buf->buffer)[15]);
851 scd->read_buffer_len = req->length;
852 svc->trace_out(scd->sd->cm, "CMIB handle_request completed");
853 svc->wake_any_pending_write(scd->conn);
854}
