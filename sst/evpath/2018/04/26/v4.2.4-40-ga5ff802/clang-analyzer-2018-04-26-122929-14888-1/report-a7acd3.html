575 sprintf(&msg_str[0], "%d", msg_limit);

598#ifdef HAVE_WINDOWS_H

618 if (result == -1) {

621 }
622 if (result == subproc_proc) {

0)) {
632 printf("Single remote subproc died with signal %d\n",

636 }

638#endif
639 if (msg_count != msg_limit) {
