648/*
649 * Create an IP socket for connection from other CMs
650 */
651extern attr_list
652libcmenet_LTX_non_blocking_listen(CManager cm, CMtrans_services svc,
653 transport_entry trans, attr_list listen_info)
654{
655 enet_client_data_ptr enet_data = trans->trans_data;
656 ENetAddress address;
657 ENetHost * server;
658
659
660 int attr_port_num = 0;
661 u_short port_num = 0;
662
663 /*
664 * Check to see if a bind to a specific port was requested
665 */
666 if (listen_info != NULL((void*)0)
667 && !query_attr(listen_info, CM_ENET_PORT,
668 NULL((void*)0), (attr_value *)(long) & attr_port_num)) {
669 port_num = 0;
670 } else {
671 if (attr_port_num > USHRT_MAX(32767 *2 +1) || attr_port_num < 0) {
672 fprintf(stderrstderr, "Requested port number %d is invalid\n", attr_port_num);
673 return NULL((void*)0);
674 }
675 port_num = attr_port_num;
676 }
677
678 svc->trace_out(cm, "CMEnet begin listen, requested port %d", attr_port_num);
679
680 address.host = ENET_HOST_ANY0;
681
682 if (enet_data->server != NULL((void*)0)) {
683 /* we're already listening */
684 if (port_num == 0) {
685 /* not requesting a specific port, return what we have */
686 return build_listen_attrs(cm, svc, NULL((void*)0), listen_info, address.port);
687 } else {
688 printf("CMlisten_specific() requesting a specific port follows other Enet operation which initiated listen at another port. Only one listen allowed, second listen fails.\n");
689 return NULL((void*)0);
690 }
691 }
692 if (port_num != 0) {
693 /* Bind the server to the default localhost. */
694 /* A specific host address can be specified by */
695 /* enet_address_set_host (& address, "x.x.x.x"); */
696
697 address.port = port_num;
698
699 svc->trace_out(cm, "CMEnet trying to bind selected port %d", port_num);
700 server = enet_host_create (& address /* the address to bind the server host to */,
701 0 /* allow up to 4095 clients and/or outgoing connections */,
702 1 /* allow up to 2 channels to be used, 0 and 1 */,
703 0 /* assume any amount of incoming bandwidth */,
704 0 /* assume any amount of outgoing bandwidth */);
705 if (server == NULL((void*)0)) {
706 fprintf (stderrstderr,
707 "An error occurred while trying to create an ENet server host.\n");
708 return NULL((void*)0);
709 }
710 enet_data->server = server;
711 } else {
712 long seedval = time(NULL((void*)0)) + getpid();
713 /* port num is free. Constrain to range 26000 : 26100 */
714 int low_bound, high_bound;
715 int size;
716 int tries;
717 srand48(seedval);
718 get_IP_config(NULL((void*)0), 0, NULL((void*)0), &low_bound, &high_bound,
719 NULL((void*)0), listen_info, svc->trace_out, (void *)cm);
720
721 restart:
722 size = high_bound - low_bound;
723 tries = 10;
724 while (tries > 0) {
725 int target = low_bound + size * drand48();
726 address.port = target;
727 svc->trace_out(cm, "CMEnet trying to bind port %d", target);
728
729 server = enet_host_create (& address /* the address to bind the server host to */,
730 0 /* 0 means dynamic alloc clients and/or outgoing connnections */,
731 1 /* allow up to 2 channels to be used, 0 and 1 */,
732 0 /* assume any amount of incoming bandwidth */,
733 0 /* assume any amount of outgoing bandwidth */);
734 tries--;
735 if (server != NULL((void*)0)) tries = 0;
736 if (tries == 5) {
737 /* try reseeding in case we're in sync with another process */
738 srand48(time(NULL((void*)0)) + getpid());
739 }
740 }
741 if (server == NULL((void*)0)) {
742 high_bound += 100;
743 goto restart;
744 }
745 enet_data->server = server;
746 }
747 svc->fd_add_select(cm, enet_host_get_sock_fd (server),
748 (select_list_func) enet_service_network, (void*)cm, (void*)trans);
749
750 svc->trace_out(enet_data->cm, "CMENET Adding read_wake_fd as action on fd %d",
751 enet_data->wake_read_fd);
752
753 svc->fd_add_select(cm, enet_data->wake_read_fd, (select_list_func)read_wake_fd_and_service,
754 (void*)cm, (void*)trans);
755
756 return build_listen_attrs(cm, svc, enet_data, listen_info, address.port);
757}

821extern int
822libcmenet_LTX_writev_func(CMtrans_services svc, enet_conn_data_ptr ecd,
823 struct iovec *iov, int iovcnt, attr_list attrs)
824{
825 int i;
826 int length = 0;
827 static time_t last_flush_call = 0;
828
829 (void) attrs;
830 for (i = 0; i < iovcnt; i++) {
831 length += iov[i].iov_len;
832 }
833
834 svc->trace_out(ecd->sd->cm, "CMENET vector write of %d bytes on peer %p",
835 length, ecd->peer);
836
837 /* Create a reliable packet of the right size */
838 ENetPacket * packet = enet_packet_create (NULL((void*)0), length,
839 ENET_PACKET_FLAG_RELIABLE);
840
841 length = 0;
842 /* copy in the data */
843 for (i = 0; i < iovcnt; i++) {
844 memcpy(packet->data + length, iov[i].iov_base, iov[i].iov_len);
845 length += iov[i].iov_len;
846 }
847
848 /* Send the packet to the peer over channel id 0. */
849 if (enet_peer_send (ecd->peer, 0, packet) == -1) {
850 enet_packet_destroy(packet);
851 svc->trace_out(ecd->sd->cm, "ENET ====== failed to send a packet to peer %p, state %d\n", ecd->peer, ecd->peer->state);
852 return -1;
853 }
854
855 wake_enet_server_thread(ecd->sd);
856
857 if (last_flush_call == 0) {
858 enet_host_flush(ecd->sd->server);
859 last_flush_call = time(NULL((void*)0));
860 } else {
861 time_t now = time(NULL((void*)0));
862 if (now > last_flush_call) {
863 last_flush_call = now;
864 enet_host_flush(ecd->sd->server);
865 }
866 }
867 return iovcnt;
868}
