20/*
21 * Some notes about the (current) operation of EVdfg.
22 *
23 * EVdfg operation is based on CM-level messages. It currently supports
24 * only one DFG per process/CM (I.E. there is no DFG identifier passed in
25 * any operation, so the DFG is implied by the contacted CM). DFGs pass
26 * through several states, including : Joining (while waiting for
27 * participating nodes), Starting (between last node join and actual
28 * operation), Running (normal operation), Reconfiguring (master changing
29 * DFG structure), and Shutdown (all nodes have voted for shutdown and
30 * we're killing the system). (State is maintained in the master, and is
31 * not reliable in the client nodes.)
32 *
33 * Generally, most operation triggers have the same sort of structure.
34 * I.E. they check to see if they are the master. If they are, the perform
35 * the operation directly, and if not they send a message to the master.
36 * The message handler then does the subroutine call that performs the
37 * operation directly. So, the message handler and the wrapper subroutine
38 * are structured similarly for most operations. However, some are handled
39 * differently. In particular, notification of client departure
40 * (connection/node failure) and client join can't be handled except when
41 * we're in the Running state. (Well, Join is handled in Joining too.)
42 * Also, voluntary reconfiguration, in addition to only being handled in
43 * the Running state, *must* be queued for later handling if it happens
44 * locally (because it's triggered by a call inside a CoD event handler,
45 * and reconfiguring while running inside a handler seems bad). So, those
46 * messages must be queued for later handling.
47 *
48 * Currently, we handle each of these messages on transition from another
49 * state into Running, or, in the case of voluntary reconfiguration if
50 * triggered locally, we handle it inside a CM delayed task (which should
51 * assure that we're at least at a message handling point).
52 */

513 src->out_count = act.u.link.port + 1;
514 }
515 in_link_found = 0;
516 for (i=0; i < dst->in_count; i++) {

518 }

523 dst->in_count = 1;
524 } else {
525 dst->in_links = realloc(dst->in_links,
526 sizeof(dst->in_links[0]) * (dst->in_count+1));
527 memset(&dst->in_links[dst->in_count], 0, sizeof(dst->in_links[0]) * (dst->in_count+1-dst->in_count));
528 dst->in_count++;
529 }
530 dst->in_links[dst->in_count - 1] = act.stone_id;
531 }
532 if (build_queue) {
533 if (src->out_links[act.u.link.port] != -1) {
534 EVdfg_config_action dact;
535 dact.type = ACT_unlink_port;
536 dact.node_for_action = src->node;
537 dact.stone_id = src->stone_id;
538 dact.u.link.port = act.u.link.port;
539 EVdfg_perform_act_on_state(config, dact, build_queue);
540 }
541 }
542 src->out_links[act.u.link.port] = act.u.link.dest_id;
543 if (build_queue) {
544 if (src->condition == EVstone_Deployed) {
545 EVdfg_config_action fact;
546 fact.type = ACT_freeze;
547 fact.stone_id = src->stone_id;
548 fact.node_for_action = src->node;
549 EVdfg_add_act_to_queue(config, fact);
550 src->condition = EVstone_Frozen;
551 }
552 EVdfg_add_act_to_queue(config, act);
553 }
554 break;
555 }
556 case ACT_link_dest: {
557 EVdfg_stone_state src = find_stone_state(act.stone_id, config);
558 EVdfg_stone_state dst = find_stone_state(act.u.link.dest_id, config);
559 int i, in_link_found, out_link_found = 0;
560 for (i=0; i < src->out_count; i++) {
561 if (src->out_links[i] == act.u.link.dest_id) out_link_found = 1;
562 }
563 if (out_link_found) {
564 return 0;
565 }
566 if (!src) {
567 return 0;
568 }
569 if (src->out_count == 0) {
570 src->out_links = malloc(sizeof(src->out_links[0]) * (src->out_count+1));
571 memset(src->out_links, 0, sizeof(src->out_links[0]) * (src->out_count+1));
572 src->out_count++;
573 } else {
574 src->out_links = realloc(src->out_links,
575 sizeof(src->out_links[0]) * (src->out_count+1));
576 src->out_count++;
577 }
578 src->out_links[src->out_count-1] = act.u.link.dest_id;
579 in_link_found = 0;
580 for (i=0; i < dst->in_count; i++) {
581 if (dst->in_links[i] == act.stone_id) in_link_found = 1;
582 }
583 if (in_link_found == 0) {
584 if (dst->in_count == 0) {
585 dst->in_links = malloc(sizeof(dst->in_links[0]));
586 memset(dst->in_links, 0, sizeof(dst->in_links[0]));
587 dst->in_count = 1;
588 } else {
589 dst->in_links = realloc(dst->in_links,
590 sizeof(dst->in_links[0]) * (dst->in_count+1));
591 memset(&dst->in_links[dst->in_count], 0, sizeof(dst->in_links[0]) * (dst->in_count+1-dst->in_count));
592 dst->in_count++;
593 }
594 dst->in_links[dst->in_count - 1] = act.stone_id;
595 }
596 if (build_queue) {
597 EVdfg_add_act_to_queue(config, act);
598 }
599 break;
600 }
601 case ACT_unlink_port: {
602 EVdfg_stone_state src = find_stone_state(act.stone_id, config);
603 EVdfg_stone_state dest;
604 if (!src) {
605 return 0;
606 }
607 if (src->out_count <= act.u.link.port) return 0;
608 if (src->out_links[act.u.link.port] == -1) {
609 /* unassigned, error to unlink */
610 return 0;
611 }
612 dest = find_stone_state(src->out_links[act.u.link.port], config);
613 if (!dest) {
614 return 0;
615 }
616 if (dest->bridge_stone) {
617 /* this stone only exists because of this link, destroy it */
618 EVdfg_config_action dact;
619 dact.type = ACT_destroy;
620 dact.stone_id = dest->stone_id;
621 EVdfg_perform_act_on_state(config, dact, build_queue);
622 }
623 src->out_links[act.u.link.port] = -1;
624 if (build_queue) {
625 EVdfg_add_act_to_queue(config, act);
626 }
627 break;
628 }
629 case ACT_unlink_dest: {
630 EVdfg_stone_state src = find_stone_state(act.stone_id, config);
631 EVdfg_stone_state dest = find_stone_state(act.u.link.dest_id, config);
632 int found = 0;
633 int i = 0;
634 if (!src) {
635 return 0;
636 }
637 if (src->out_count <= act.u.link.port) return 0;
638 for (i=0; i < src->out_count; i++) {
639 if (src->out_links[i] == dest->stone_id) {
640 /* remove this, move remaining down */
641 memmove(&src->out_links[i], &src->out_links[i+1],
642 sizeof(src->out_links[0]) * (src->out_count - i - 1));
643 found++;
644 } else {
645 EVdfg_stone_state possible_bridge = find_stone_state(src->out_links[i], config);
646 if (possible_bridge->bridge_stone && (possible_bridge->out_links[0] == dest->stone_id)) {
647 /* there was a bridge stone in between, but we found the link */
648 /* the bridge stone only exists because of this link, destroy it */
649 EVdfg_config_action dact;
650 dact.type = ACT_destroy;
651 dact.stone_id = possible_bridge->stone_id;
652 EVdfg_perform_act_on_state(config, dact, build_queue);
653 /* remove this link, move remaining down */
654 memmove(&src->out_links[i], &src->out_links[i+1],
655 sizeof(src->out_links[0]) * (src->out_count - i - 1));
656 found++;
657 /* rewrite action to refer to link by position */
658 act.type = ACT_unlink_port;
659 act.u.link.port = i;
660 }
661 }
662 }
663 if (found == 0) {
664 /* not found, error to unlink */
665 return 0;
666 }
667 if (build_queue) {
668 EVdfg_add_act_to_queue(config, act);
669 }
670 break;
671 }
672 case ACT_set_attrs: {
673 EVdfg_stone_state stone = find_stone_state(act.stone_id, config);
674 if (!stone) {
675 return 0;
676 }
677 if (stone->attrs != NULL((void*)0)) {
678 free_attr_list(stone->attrs);
679 }
680 stone->attrs = act.u.attrs.attrs;
681 if (build_queue) {
682 EVdfg_add_act_to_queue(config, act);
683 }
684 break;
685 }
686 case ACT_assign_node: {
687 EVdfg_stone_state stone = find_stone_state(act.stone_id, config);
688 if (!stone) {
689 return 0;
690 }
691 stone->node = act.u.assign.dest_node;
692 if (build_queue) {
693 EVdfg_add_act_to_queue(config, act);
694 }
695 break;
696 }
697 case ACT_destroy: {
698 EVdfg_stone_state stone = find_stone_state(act.stone_id, config);
699 if (!stone) {
700 return 0;
701 }
702 if (build_queue) {
703 EVdfg_add_act_to_queue(config, act);
704 }
705 break;
706 }
707 default:
708 printf("Bad action in perform_act_on_state %s (%d)\n", ACT_string[act.type], act.type);
709 return 0;
710 break;
711 }
712 return 1;
713}
