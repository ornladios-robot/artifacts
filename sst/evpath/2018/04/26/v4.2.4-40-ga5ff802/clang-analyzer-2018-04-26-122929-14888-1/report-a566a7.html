645libcmudp_LTX_writev_func(svc, ucd, iov, iovcnt, attrs)
646CMtrans_services svc;
647udp_conn_data_ptr ucd;
648struct iovec *iov;
649int iovcnt;
650attr_list attrs;
651{
652 int fd = ucd->utd->socket_fd;
Value stored to 'fd' during its initialization is never read
653 struct sockaddr_in addr = ucd->dest_addr;
654 struct msghdr msg;
655 if (ucd->utd->socket_fd == -1) {
656 if ((ucd->utd->socket_fd = socket(AF_INET2, SOCK_DGRAMSOCK_DGRAM, 0)) < 0) {
657 perror("socket");
658 exit(1);
659 }
660 }
661 fd = ucd->utd->socket_fd;
662 svc->trace_out(ucd->utd->cm, "CMUdp writev of %d vectors on fd %d",
663 iovcnt, fd);
664 memset(&msg, 0, sizeof(msg));
665 msg.msg_name = (void*)&addr;
666 msg.msg_namelen = sizeof(addr);
667 msg.msg_iov = &iov[0];
668 msg.msg_iovlen = iovcnt;
669 if (sendmsg(fd, &msg, 0) < 0) {
670 perror("write sendmsg");
671 exit(1);
672 }
673 return iovcnt;
674}

691extern void *
692libcmudp_LTX_initialize(cm, svc)
693CManager cm;
694CMtrans_services svc;
695{
696 static int atom_init = 0;
697
698 udp_transport_data_ptr udp_data;
699 svc->trace_out(cm, "Initialize CMUdp transport");
700#ifdef HAVE_WINDOWS_H
701 if (socket_global_init == 0) {
702 int nErrorStatus;
703 /* initialize the winsock package */
704 nErrorStatus = WSAStartup(wVersionRequested, &wsaData);
705 if (nErrorStatus != 0) {
706 fprintf(stderrstderr, "Could not initialize windows socket library!");
707 WSACleanup();
708 exit(-1);
709 }
710 }
711#endif
712 if (atom_init == 0) {
713 CM_UDP_PORT = attr_atom_from_string("UDP_PORT");
714 CM_UDP_ADDR = attr_atom_from_string("UDP_ADDR");
715 CM_IP_HOSTNAME = attr_atom_from_string("IP_HOST");
716 CM_TRANSPORT = attr_atom_from_string("CM_TRANSPORT");
717 CM_TRANSPORT_RELIABLE = attr_atom_from_string("CM_TRANSPORT_RELIABLE");
718 atom_init++;
719 }
720 udp_data = svc->malloc_func(sizeof(struct udp_transport_data));
721 udp_data->cm = cm;
722 udp_data->svc = svc;
723 udp_data->socket_fd = -1;
724 udp_data->self_ip = 0;
725 udp_data->self_port = 0;
726 udp_data->connections = NULL((void*)0);
727 udp_data->characteristics = create_attr_list();
728 add_int_attr(udp_data->characteristics, CM_TRANSPORT_RELIABLE, 0);
729 svc->add_shutdown_task(cm, free_udp_data, (void *) udp_data, FREE_TASK2);
730 return (void *) udp_data;
731}
