684/***********************************************************
685Name: AtA
686Passed: **A, a matrix of dim m by n
687Returns: **R, a matrix of dim m by p
688***********************************************************/
689static void
690AtA(double **A, int m, int n, double **R)
691{
692 int i, j, k;
693
694 for (i = 0; i < n; i++)
695 for (j = 0; j < n; j++) {
696 R[i][j] = 0.0;
697 for (k = 0; k < m; k++)
698 R[i][j] += A[k][i] * A[k][j];
699 }
700}

702/***********************************************************
703Name: Atf
704Passed: **A, a matrix of dim n by p
705*f, a vector of dim n
706Returns: **R, a matrix of dim p by 1
707***********************************************************/
708static void
709Atf(double **A, double *f, int n, int p, double **R)
710{
711 int i, j, k;
712
713 for (i = 0; i < p; i++)
714 for (j = 0; j < 1; j++) {
715 R[i][j] = 0.0;
716 for (k = 0; k < n; k++)
717 R[i][j] += A[k][i] * f[k];
718 }
719}
720/*********************************************************************
721 * dp_inv: inverse matrix A
722 * matrix A is replaced by its inverse
723 * returns ks, an integer
724 * A - square matrix of size n by n
725 * ks equals n when A is invertable
726 * example usage:
727 if(dp_inv(N,u) != u){
728 printf("Error inverting N\n");
729 exit(-1);}
730***********************************************************/
731static int
732dp_inv(double **A, int n)
733{
734 int i, j, k, ks;
735
736 ks = 0;
737 for (k = 0; k < n; k++) {
738 if (A[k][k] != 0.0) {
739 ks = ks + 1;
740 for (j = 0; j < n; j++) {
741 if (j != k)
742 A[k][j] = A[k][j] / A[k][k];
743 } /* end for j */
744 A[k][k] = 1.0 / A[k][k];
745 } /* end if */
746 for (i = 0; i < n; i++) {
747 if (i != k) {
748 for (j = 0; j < n; j++) {
749 if (j != k)
750 A[i][j] = A[i][j] - A[i][k] * A[k][j];
751 } /* end for j */
752 A[i][k] = -A[i][k] * A[k][k];
753 } /* end if */
754 } /* end for i */
755 } /* end for k */
756 return ks;
757} /* end MtxInverse() */

774 for (k = 0; k < m; k++)
